CREATE TABLE TBVENDAS(
NUMVENDA NUMBER PRIMARY KEY, 
CODFUNC NUMBER NOT NULL,
CODCLIE NUMBER,
DATAEMISSAO DATE NOT NULL,
SITUACAO VARCHAR(20) NOT NULL,
TIPO VARCHAR(10) NOT NULL,
SUBTOTAL NUMBER,
DESCONTOS NUMBER,
CONDPAGTO VARCHAR(9) NOT NULL,
FORMAPAGTO VARCHAR(9) NOT NULL,
JUROS NUMBER,
DATAENTREGA DATE,
FRETE NUMBER,
PARCELASRES NUMBER,
DESCRICAO VARCHAR(150),
CONSTRAINT FKCODFUNC_VENDAS FOREIGN KEY (CODFUNC) REFERENCES TBFUNCIONARIO (CODFUNC),
CONSTRAINT FKCODCLIE_VENDAS FOREIGN KEY (CODCLIE) REFERENCES TBCLIENTE (CODCLIE));

CREATE TABLE TBITENSVENDA(
CODITEM NUMBER PRIMARY KEY, 
NUMVENDA NUMBER NOT NULL,
CODPRODUTO NUMBER NOT NULL,
QTDADE NUMBER NOT NULL,
CONSTRAINT FKNUMVENDA FOREIGN KEY (NUMVENDA) REFERENCES TBVENDAS (NUMVENDA),
CONSTRAINT FKCODPRODUTO FOREIGN KEY (CODPRODUTO) REFERENCES TBPRODUTO (CODPRODUTO));
